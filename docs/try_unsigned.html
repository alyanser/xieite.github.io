xieite::try_unsigned<>

Evaluates to the unsigned variant of a type, if it exists.

template<typename T>
using try_unsigned = std::conditional_t<std::signed_integral<T>, std::make_unsigned<T>, std::type_identity<T>>::type;


Example

import std;
import xieite;

int main() {
    std::print("{}\n", std::same_as<unsigned int, xieite::try_unsigned<int>>);
    std::print("{}\n", std::same_as<double, xieite::try_unsigned<double>>);
}
