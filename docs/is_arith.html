xieite::is_arith<>

Specifies that a type is arithmetic (integral or floating-point).

template<typename T>
concept is_arith = std::integral<T> || std::floating_point<T>;


Example

import std;
import xieite;

template<xieite::is_arith T>
T add(T a, T b) {
    return a + b;
}

int main() {
    std::print("{}\n", add(2, 3));
    std::print("{}\n", add(5.1, 4.6));
}
5
9.7
