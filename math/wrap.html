xieite::wrap()

 Defined in fragment xieite:wrap

Takes the modulo of a number between two limits, inclusively.


Declaration
template<xieite::is_arith T>
[[nodiscard]] constexpr T wrap(T value, std::type_identity_t<T> limit0, std::type_identity_t<T> limit1) noexcept;


Example
import xieite;

int main() {
    for (int i = 0; i < 10; ++i) {
        xieite::dump(xieite::wrap(i, 4, -2));
    }
}
Output:
0
1
2
3
4
-2
-1
