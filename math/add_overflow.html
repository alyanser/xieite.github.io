xieite::add_overflow()

Computes whether the addition of two or more integers would overflow or underflow their type (thereby invoking undefined behavior with signed integers).

template<xieite::is_arith T, std::convertible_to<T>... Ts>
[[nodiscard]] constexpr bool add_overflow(T first, Ts... rest) noexcept;


Example

import std;
import xieite;

int main() {
    std::print("{}\n", xieite::add_overflow<std::int8_t>(127, 1));
}
true
